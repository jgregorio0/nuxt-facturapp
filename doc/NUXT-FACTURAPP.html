1 Requerimientos

1.1 Node JS y NPM con NVM

$ git clone https://github.com/creationix/nvm.git ~/.nvm && cd ~/.nvm && git checkout `git describe --abbrev=0 --tags`
$ echo "source ~/.nvm/nvm.sh" >> ~/.profile
$ source ~/.profile
$ nvm install 8.11.4
$ nvm alias default 8.11.4
$ node -v
$ npm -v

5 Store

Con Nuxt el módulo vuex-persistedstate debe ser incluido como plugin:

// nuxt.config.js


...
plugins: [{ src: '~/plugins/localStorage.js', ssr: false }]
...
// ~/plugins/localStorage.js


import createPersistedState from 'vuex-persistedstate'


export default ({store}) => {
createPersistedState({
key: 'yourkey',
paths: [...]
...
})(store)
}

6 Bootstrap

  • nuxt.config.js

const webpack = require('webpack')


module.exports = {
build: {
vendor: ['jquery', 'bootstrap'],
plugins: [
// set shortcuts as global for bootstrap
new webpack.ProvidePlugin({
$: 'jquery',
jQuery: 'jquery',
'window.jQuery': 'jquery'
})
]
},
// include bootstrap css
css: ['bootstrap/dist/css/bootstrap.css'],
// include bootstrap js on startup
plugins: ['~plugins/bootstrap.js']
}


  • plugins/bootstrap.js

// Include bootstrap JS only
if (process.BROWSER_BUILD) {
require('bootstrap')
}

7 Bootstrap-vue

  1. Instalar dependencia boostrap-vue

npm i bootstrap-vue




  1. Anadir bootstrap como modulo en la configuración de Nuxt


  • nuxt.config.js

{
modules: [
'bootstrap-vue/nuxt',


// Or if you have custom bootstrap CSS...
['bootstrap-vue/nuxt', { css: false }],
]
}



8 Font awesome

  1. Incluir dependencia del módulo font-awesome por encima de nuxt

"dependencies": {
"@fortawesome/free-solid-svg-icons": "^5.3.1",
"nuxt-fontawesome": "^0.3.0",
"nuxt": "latest"
},



En caso de que build falle añadir:

  • // "@fortawesome/fontawesome-svg-core": "^1.2.4",

  • // "@fortawesome/vue-fontawesome": "^0.1.1",




  1. Anadirlo al fichero nuxt.config.js

    • Se puede importar el conjunto completo o solo unos iconos

/*
** Modules
*/
modules: [""nuxt-fontawesome"],
/*
** fontawesome
*/
fontawesome: {
component: "fa",
imports: [
{
set: "@fortawesome/free-solid-svg-icons",
icons: [
"faSave",
"faFolderOpen",
"faTrash",
"faBed",
"faReceipt",
"faDollarSign",
"faChartLine"
]
}
]
},



  1. Para utilizarlo

<template>
<div>
<fa :icon="fas.faAddressBook" />
<fa :icon="faGithub" />
</div>
</template>
<script>
import { fas } from '@fortawesome/free-solid-svg-icons'
import { faGithub } from '@fortawesome/free-brands-svg-icons'
export default {
computed: {
fas () {
return fas
},
faGithub () {
return faGithub
}
},
}
</script>
